Step 1)
#Create the puthon env 
python3 -m venv vllm_venv

source vllm_venv/bin/activate

pip install -r requirements.txt


Step 2)
#Get k8s objects ready
kubectl apply -f configmap.yml
kubectl apply -f service.yml
kubectl apply -f deploy.yml

Step 3)
#We expose this service we want to use local host so port forward localhost to service
kubectl port-forward svc/vllm-api 8000:8000 &

Step 4)
Test curl first 

curl -s http://localhost:8000/v1/models | jq


Step 5)
To setup the GUI run below
uvicorn ui:app --host 0.0.0.0 --port 8080
